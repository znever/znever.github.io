$(document).ready(function(){
	$("input").click(function(){
		var _data = {};
		_data.pageNum = 1;
		_data.pageSize = 20;
		
		var $userN = $("#username").val();
		if($userN != ""){
			_data.name = $userN;
		}
		
		var $phone = $("#phone").val();
		if($phone != ""){
			_data.mobile = $phone;
		}
		
		var $college = $("#college").val();
			if($college != ""){
			_data.college = $college;
		}
		
		$.ajax({
			type:"get",
			url:"http://www.hemijiaoyu.com/service/searchApply",
			dataType:"JSON",
			data:_data,
			success:function(msg){
				var arr = msg.list;
				var str = "";
				for(var i = 0; i < arr.length; i++){
					str += "";
					var obj = arr[i];		
					for(var j in arr[i]){
									


if(j != "homeTel"){
	str += "<"td style='border-top:1px solid #666;
		border-right:none;border-left:none;height:40px;
	text-align:center;'>" + obj[j] + "<"/td>";
						}
									
					}
					str += "<""/tr>";			
				}
$("#tbd").html(str);
		$("tr").dblclick(function(e){
		$("#mask").show();
								
		var i = e.currentTarget.id;
		$("#name").val(arr[i].name);
		$("#sex").val(arr[i].sex);
		$("#birth").val(arr[i].birthday);
		$("#mobile").val(arr[i].mobile);
		$("#political").val(arr[i].political);
		$("#qq").val(arr[i].qq);
		$("#college").val(arr[i].college);
		$("#faculty").val(arr[i].faculty);
		$("#major").val(arr[i].major);
		$("#father").val(arr[i].fatherName);
		$("#fatherMobile").val(arr[i].fatherMobile);
		$("#mother").val(arr[i].matherName);
		$("#motherMobile").val(arr[i].matherMobile);
		$("#address").val(arr[i].homeAddress);
		$("#information").val(arr[i].source);
		$("#majorDes").val(arr[i].majorMemo);
		$("#selfDes").val(arr[i].selfMemo);
		console.log(arr[i]);
		});
		$("#cha").click(function(){
		$("#mask").hide();
		});
		},
		error:function(error){
			console.log("错误:" + error);
						}
					});
