图片跟着鼠标走

clientX属性：返回当事件被触发时，鼠标指针的水平坐标。
clientY属性：返回当事件被触发时，鼠标指针的垂直坐标。
	“<”p>鼠标X: “<”span id="sp1">“<”/span>“<”/p>
		“<”p>鼠标Y: “<”span id="sp2">“<”/span>“<”/p>
		“<”img src="img/01.gif" style="width:50px; 
		height:50px; position:absolute; left:0; top:0;" />
		
	“<”script>
		window.addEventListener("mousemove",moveHandle);
		var sp1 = document.getElementById("sp1");
		var sp2 = document.getElementById("sp2");
		var img = document.getElementsByTagName('img');
			
		function moveHandle(e){
			e.preventDefault();
			sp1.innerText = e.clientX;
			sp2.innerText = e.clientY;
			img[0].style.left = e.clientX - img[0].width/2 + "px";
			img[0].style.top = e.clientY - img[0].height/2 + "px";
		}
	“<”/script>

获取手机验证码

	“<”button id="btn">获取手机验证码“<”/button>
		
		“<”script>
			var btn = document.getElementById('btn');
			
			btn.addEventListener("click",clickHander,false);
			var t = 10;
			
			function clickHander(){
				btn.innerHTML = t + "秒后重新获取";
				btn.disabled = true;
				siv1 = window.setInterval(updateTime,1000);
			}
			
			function updateTime(){
				btn.innerHTML = t + "秒后重新获取";
				if(t <= 0){
					if(siv1){
						clearInterval(siv1);
					}
						t = 10;
						btn.innerHTML = "获取手机验证码";
						btn.disabled = false;
					}
					t--;
				}
			
		“<”/script>
